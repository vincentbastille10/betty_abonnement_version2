<!-- Betty Bot — snippet (injection d’iframe) -->
<div id="bettybot-container" style="position:relative; width:100%; max-width:420px; height:620px; margin:0 auto"></div>
<script>(function(){
  var pid = "{{ public_id }}";
  var buyer = "{{ buyer_email|default('', true)|urlencode }}";
  var src = "{{ base_url }}/chat?public_id=" + encodeURIComponent(pid) + "&embed=1" + (buyer ? "&buyer_email=" + buyer : "");
  var root = document.getElementById("bettybot-container");
  var f = document.createElement("iframe");
  f.src = src;
  f.title = "{{ full_name }}";
  f.style.cssText = "width:100%;height:100%;border:0;border-radius:16px;box-shadow:0 10px 30px rgba(0,0,0,.25);background:#0b0f1e;";
  f.setAttribute("loading","lazy");
  f.setAttribute("referrerpolicy","no-referrer-when-downgrade");
  f.setAttribute("allow","clipboard-read; clipboard-write; microphone; autoplay");
  root.appendChild(f);
})();</script>
